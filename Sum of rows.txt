#include<iostream>
using namespace std;
int main()
{
  int m,n;
  std::cin>>m>>n;
  int mat[m][n];
  for(int i=0;i<m;i++)
    for(int j=0;j<n;j++)
      std::cin>>mat[i][j];
  for (int i = 0; i < m; i++) 
  {
    int sum = 0;
    for (int j = 0; j < n; j++) 
    {
      sum += mat[i][j];
    } 
    std::cout<<sum<<"\n";
  }
}
Sum of rows
Seenu have a fruit shop and arranged the some set of fruits in column and row wise. Seenu needs to find the total number of fruits in each rows. Help him to find out.
INPUT & OUTPUT FORMAT:

Input consists of 2 integers and 1 2D-array. Integers correspond to the size of rows and columns.

SAMPLE INPUT & OUTPUT:

3

3

1 2 3

7 3 1

7 4 1

6

11

12
